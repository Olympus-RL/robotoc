macro(add_benchmark BENCHMARK)
  add_executable(
    ${BENCHMARK} 
    ${BENCHMARK}.cpp
  )
  target_link_libraries(
    ${BENCHMARK} 
    PRIVATE
    robotoc::robotoc
  )
endmacro()

macro(add_example EXAMPLE)
  add_executable(
    ${EXAMPLE} 
    ${EXAMPLE}.cpp
  )
  target_link_libraries(
    ${EXAMPLE} 
    PRIVATE
    robotoc::robotoc
  )
endmacro()
